    <script>

        // --- 📝 EDIT YOUR COMPLIMENTS HERE ---

        const compliments = [

            "Your smile effortlessly makes my entire day brighter. 😊",

            "I am so incredibly proud of you and how hard you work.",

            "You are my favorite person to absolute do nothing with. 🛋️",

            "Just a reminder: you look incredibly handsome today.",

            "Thank you for being my safe space and my biggest comfort.",

            "My favorite place in the world is right next to you. 🌎",

            "You have the kindest heart, and I'm so lucky you share it with me.",

            "You are my absolute best friend and my favorite plot twist. 💖",

            "Even when things get messy, you're the one I want by my side.",

            "You make me laugh harder than anyone else can! 😂"

        ];


        const btn = document.getElementById('loveBtn');

        const display = document.getElementById('complimentDisplay');

        const meter = document.getElementById('loveMeter');

        const label = document.getElementById('meterLabel');

        

        let clicks = 0;

        let lastIndex = -1;


        btn.addEventListener('click', () => {

            // 1. Select a random compliment (without repeating the immediate last one)

            let randomIndex;

            do {

                randomIndex = Math.floor(Math.random() * compliments.length);

            } while (randomIndex === lastIndex && compliments.length > 1);

            

            lastIndex = randomIndex;


            // 2. Display text with pop animation

            display.classList.remove('fade-in');

            void display.offsetWidth; // Trigger CSS reflow

            display.textContent = compliments[randomIndex];

            display.classList.add('fade-in');


            // 3. Update Love Meter

            clicks++;

            let progress = Math.min(5 + (clicks * 10), 100);

            meter.style.width = progress + '%';

            

            if (progress >= 100) {

                label.textContent = "Love Level: Overflowing! 🎉❤️";

            } else {

                label.textContent = `Love Level: ${progress}%`;

            }


            // 4. Spawn Heart Confetti

            createConfetti();

        });


        function createConfetti() {

            const btnRect = btn.getBoundingClientRect();

            

            // Create 8 little floating hearts per click

            for (let i = 0; i < 8; i++) {

                const heart = document.createElement('div');

                heart.className = 'confetti';

                heart.innerHTML = ['❤️', '💖', '✨', '🌸'][Math.floor(Math.random() * 4)];

                

                // Position around the button

                heart.style.left = (btnRect.left + (btnRect.width / 2) + (Math.random() * 100 - 50)) + 'px';

                heart.style.top = (btnRect.top + 'px');

                heart.style.fontSize = (Math.random() * 10 + 14) + 'px';

                

                // Random horizontal drift

                heart.style.setProperty('--drift', (Math.random() * 100 - 50) + 'px');

                

                document.body.appendChild(heart);

                

                // Clean up elements after animation ends

                setTimeout(() => heart.remove(), 1200);

            }

        }
